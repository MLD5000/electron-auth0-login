const defaultAdapters = adapters(
  tokenStore,
  auth0LoginPage,
  auth0LogoutPage,
  auth0HttpAPI,
  logging
);

export const auth = library(defaultAdapters);

export const authWithKeytar = (keytar) => library(
  adapters(
    defaultAdapters,
    keytarRefreshTokens(keytar)
  )
);

export const customLogin = (loginPageOptions) => library(
  adapters(
    defaultAdapters,
    auth0LoginPage(loginPageOptions)
  )
)
